Delete duplicates and Sort

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,i=0,b,j,t,k,b1;

int a[200];

while(scanf("%d ",&N)>0)

{

a[i]=N;

i++;

}

for(b=0;b< i;b++)

{

for(j=b+1;j< i;j++)

{

if(a[b]==a[j])

{

for(k=j;k< i;k++)

{

a[k]=a[k+1];

i--;

j--;

}

}

}

for(b=0;b< i;b++)

{

for(b1=b+1;b1< i;b1++)

{

if(a[b]>a[b1])

{

t=a[b];

a[b]=a[b1];

a[b1]=t;

}

}

}

for(b=0;b< i;b++)

{

printf("%d ",a[b]);

}

}

input:

50 60 70 80 90 20 60

output:

20 50 60 70 80 90
