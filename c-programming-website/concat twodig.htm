Concatenate two Digits

#include< stdio.h>

#include< stdlib.h>

int main()

{

long long int M,N,p,q,k;

scanf("%lld %lld",&M,&N);

k=M%10;

if(k<0)

{

if(-k==N)

printf("%lld",M/10);

else

printf("%lld",(M*10)-N);

}

else if(k>0)

{

if(k==N)

printf("%lld",M/10);

else

printf("%lld",(M*10)+N);

}

else if(k==0)

{

if(M<0)

{

if(k!=N)

printf("%lld",(M*10)-N);

else

printf("%lld",M/10);

}

else

{

if(k!=N)

printf("%lld",(M*10)+N);

else

printf("%lld",M/10);

}

}

}

100 3

103
