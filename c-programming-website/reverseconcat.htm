Reverse and concatenate sum

#include< stdio.h>

#include< stdlib.h>

int main()

{

long long int N,d,S=0,i,a,b=0;

scanf("%lld",&N);

while(N)

{

d=N%10;

S=0;

for(i=d;i>=0;i--)

{

S=S*10+i;

}

b=b+S;

printf("%lld ",b);

N=N/10;

}

}

input:

12345

output:

543210 586420 589630 589840 589850
