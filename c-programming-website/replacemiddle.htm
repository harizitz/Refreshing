#include< stdio.h>

#include< stdlib.h>

int main()

{

char S[1000];

scanf("%s",S);

int i,len=strlen(S),d,e;

d=len/2;

e=(len/2)-1;

if(len%2==1){

for(i=0;i< len;i++)

{

if(i==d)

printf("*");

else

printf("%c",S[i]);

}}

else{

for(i=0;i< len;i++)

{

if(i==d||i==e)

printf("*");

else

printf("%c",S[i]);

}}

}

harish
