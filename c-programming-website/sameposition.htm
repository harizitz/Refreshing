Same position Elements

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,i,j;

scanf("%d",&N);

int a[N],b[N];

for(i=0;i< N;i++)

{

scanf("%d ",&a[i]);

}

for(j=0;j< N;j++)

{

scanf("%d ",&b[j]);

}

for(i=0,j=0;i< N,j< N;i++,j++)

{

printf("%d %d\n",a[i],b[j]);

}

}

input:

3

1 2 3

4 5 6

output:

1 4

2 5

3 6
