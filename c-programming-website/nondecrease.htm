First half and second half sum

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,i,d,f=0;

scanf("%d",&N);

int a[N];

for(i=0;i< N;i++)

{

scanf("%d ",&a[i]);

}

d=a[0];

for(i=1;i< N;i++)

{

if(d>a[i])

{

f++;

}

else

{

f=0;

}

}

if(f==N-1)

{

printf("Invalid");

}

else

{

printf("Valid");

}

}

input:

5

1 2 3 4 5

output:

Valid
