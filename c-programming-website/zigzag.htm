#include< stdio.h>

#include< stdlib.h>

int main()

{

char S[1000];

scanf("%s",S);

int n=strlen(S),i,j,p=0,p1=n-1;

for(i=1;i<=n;i++)

{

for(j=1;j<=n;j++)

{

if(i==j)

printf("%c",S[p]);

else if(i+j==n+1)

printf("%c",S[p1]);

else if(i!=j||i+j!=n+1)

printf(" ");

}

p1--;

p++;

printf("\n");

}

}
