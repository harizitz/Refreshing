Upper right diagonal matrix

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,i,j;

scanf("%d",&N);

int a[N][N];

for(i=0;i< N;i++)

{

for(j=0;j< N;j++)

{

scanf("%d ",&a[i][j]);

}

}

for(i=0;i< N;i++)

{

for(j=0;j< N;j++)

{

if(i>j)

printf("* ");

else

printf("%d ",a[i][j]);

}

printf("\n");

}

}

input:

3

1 2 3

4 5 6

7 8 9

output:

1 2 3

* 5 6

* * 4
