First X and Last X Sum

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,X,i,S=0,S1=0,j;

scanf("%d\n",&N);

int a[N];

for(i=0;i< N;i++)

{

scanf("%d ",&a[i]);

}

scanf("%d",&X);

for(i=0;i< N;i++)

{

if(i< X)

{

S=S+a[i];

}

}

for(j=N-1;j>=N-X;j--)

{

S1=S1+a[j];

}

printf("%d",S+S1);

}

input:

5

1 2 3 4 5

2

output:

12
