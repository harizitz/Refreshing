Negative digits in Array

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,i,f=0;

scanf("%d",&N);

int a[N];

for(i=0;i< N;i++)

{

scanf("%d ",&a[i]);

}

for(i=N-1;i>=0;i--)

{

if(a[i]< 0)

{

printf("%d ",a[i]);

}

else

{

f++;

}

}

if(f==N)

{

printf("-1");

}

}

input:

5

-2 4 5 3 -4

output:

-4 -2
