Division of two factorial Numbers

#include< stdio.h>

#include< stdlib.h>

int main()

{

long long int A,B,f=1,g;

scanf("%lld %lld",&A,&B);

g=A;

if(g< 10&&B< 10)

{

while(g>=B)

{

f=f*g;

g--;

}

printf("%lld",f/B);

}

else

{

while(g>B)

{

f=f*g;

g--;

}

printf("%lld",f);

}

}

4 2

12
