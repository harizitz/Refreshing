  10. Half reverse pyramid

    11. #include< stdio.h>

    #include < stdlib.h>

    int main()

    {

    int N,i,j,p;

    scanf("%d",&N);

    for(i=1;i<=N;i++)

    {

    for(j=1;j<=i;j++)

    {

    p=i-j+1;

    printf("%d ",p);

    }

    printf("\n");

    }

    }

    input:

    5

    output:

    1

    2 1

    3 2 1

    4 3 2 1

    5 4 3 2 1
