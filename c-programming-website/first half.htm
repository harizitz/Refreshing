First half and second half sum

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,c=0,i=0,j,d,S=0,S1=0;

int a[1000];

while(scanf("%d ",&N)>0)

{

a[i]=N;

i++;

N=N/10;

}

for(j=0;j< i;j++)

{

c++;

}

d=c/2;

for(j=0;j< i;j++)

{

if(j< d)//first half condition

S=S+a[j];//sum

else// second half

S1=S1+a[j];//sum

}

printf("%d",S*S1);//first half sum * second half sum

}

input:

1 2 3 4

output:

21
