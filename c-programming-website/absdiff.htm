Absolute max difference of Number

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,M1=0,i,X,j;

scanf("%d",&N);

int a[N];

for(i=0;i< N;i++)

{

scanf("%d ",&a[i]);

}

for(i=0,j=i+1;i< N,j< N;i++,j++)

{

X=abs(a[i]-a[j]);

if(X>M1)

{

M1=X;

}

}

printf("%d",M1);

}

input:

5

10 20 30 40 60

output:

20
