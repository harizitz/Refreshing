Matrix Max 1's count

#include< stdio.h>

#include< stdlib.h>

int main()

{

int R,C,c1=0,i,j,max=0;

scanf("%d %d",&R,&C);

int a[R][C];

for(i=0;i< R;i++)

{

for(j=0;j< C;j++)

scanf("%d ",&a[i][j]);

}

for(i=0;i< R;i++)

{ c1=0;

for(j=0;j< C;j++)

{

if(a[i][j]==1)

c1++;

if(c1>max)

max=c1;

}

}

printf("%d",max);

}
