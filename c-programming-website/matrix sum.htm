Matrix sum of x row and column

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N;

scanf("%d",&N);

int i,j,a[N+1][N+1],X,S=0,S1=0;

for(i=1;i<=N;i++)

{

for(j=1;j<=N;j++)

{

scanf("%d ",&a[i][j]);

}

}

scanf("%d",&X);

for(i=1;i<=N;i++)

{

for(j=1;j<=N;j++)

{

if(i==X)

{

S=S+a[i][j];

}

if(j==X)

{

S1=S1+a[i][j];

}

}

}

printf("%d %d",S,S1);

}
