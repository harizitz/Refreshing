#include< stdio.h>

#include< stdlib.h>

int main()

{

long long int N,s,l,b,t1,t2,t3;

scanf("%lld",&N);

if(N==1)

{

scanf("%lld",&s);

printf("%lld",s*4);

}

else if(N==2)

{

scanf("%lld %lld",&l,&b);

printf("%lld",2*(l+b));

}

else if(N==3)

{

scanf("%lld %lld %lld",&t1,&t2,&t3);

printf("%lld",t1+t2+t3);

}

else

{

printf("-1");

}

}
