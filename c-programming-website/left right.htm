Left and Right Diagonal Sum

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N;

scanf("%d",&N);

int a[N][N],i,j,S=0,S1=0;

for(i=0;i< N;i++)

{

for(j=0;j< N;j++)

scanf("%d ",&a[i][j]);

}

for(i=0;i< N;i++)

{

for(j=0;j< N;j++)

if(i==j)

S=S+a[i][j];

}

for(i=0;i< N;i++)

{

for(j=0;j< N;j++)

if(i+j==N-1)

S1=S1+a[i][j];

}

printf("%d",S*S1);

}

3

1 2 3

4 5 6

7 8 9

output:

225
