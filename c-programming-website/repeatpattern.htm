Integer repeat pattern

#include< stdio.h>

#include < stdlib.h>

int main()

{

int N,i,j,p,k,a,b,d;

scanf("%d",&N);

p=N,k=N;

a=N+1,b=N+2,d=1;

for(i=1;i<=N+2;i++)

{

for(j=1;j<=N+1;j++)

{

if(i==N+2)

{

printf("%d",k);

}

else if(j==a-1&&i!=1&&i!=b)

{

printf("%d",d);

d++;

}

else

{

printf("%d",p);

}

}

p++;

printf("\n");

}

}
