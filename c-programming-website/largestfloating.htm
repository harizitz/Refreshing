Largest Floating point Numbers

#include< stdio.h>

#include< stdlib.h>

int main()

{

float A,B,C,p,i,j,k;

scanf("%f %f %f",&A,&B,&C);

i=A-trunc(A);

//trunc removes point after numbers

j=B-trunc(B);

k=C-trunc(C);

if(j>i&&j>k)

{

printf("%.2f",B);

}

else if(i>j&&i>k)

{

printf("%.2f",A);

}

else if(k>i&&k>j)

{

printf("%.2f",C);

}

}

input:

19.34 12.98 10.45

output:

12.98
