Sum and product equal to Digit

#include< stdio.h>

#include< stdlib.h>

int main()

{

int R,C,i,j;

scanf("%d %d\n",&R,&C);

char a[R][C];

for(i=0;i< R;i++)

{

for(j=0;j< C;j++)

{

scanf("%c ",&a[i][j]);

}

}

for(i=0;i< R;i++)

{

for(j=0;j< C;j++)

}

printf("%c ",a[i][j]);

}

printf("\n");

}

}

input:

3 3

a b c

d e f

g h i

output:

a b c

d e f

g h i
