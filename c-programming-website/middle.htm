Sorted array Middle Element

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N;

scanf("%d",&N);

int a[N],t,i,j;

for(i=0;i< N;i++)

{

scanf("%d ",&a[i]);

}

for(i=0;i< N;i++)

{

for(j=i+1;j< N;j++)

if(a[i]>a[j])

{

t=a[i];

a[i]=a[j];

a[j]=t;

}

}

if(N%2!=0)

{

printf("%d ",a[N/2]);

}

else

{

printf("%d %d",a[N/2-1],a[N/2]);

}

}
