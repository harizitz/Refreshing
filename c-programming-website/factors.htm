Factors of Array Element

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N;

scanf("%d",&N);

int a[N],i,j;

for(i=0;i< N;i++)

{

scanf("%d",&a[i]);

for(j=1;j<=1000;j++)

{

if(a[i]%j==0)

{

printf("%d ",j);

}}

printf("\n");

}

}

input:

3

10 20 30

output:

1 2 5 10

1 2 4 5 10 20

1 2 3 5 6 10 15 30
