Convert Seconds into Days,Hours,Minutes

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,s,m,h,d,h1,m1,h2,h3;

scanf("%d",&N);

d=(N/86400);

printf("%dD ",d);

h=(N/60);

h1=h/60;

h2=h1%24;

if(h2>12)

{

h3=h2/12;

}

else

{

h3=h2;

}

m=N/60;

s=N%60;

if(m>60)

{

m1=m%60;

}

else if(m<=60)

{

m1=m;

}

printf("%02d:%02d:%02d",h3,m1,s);

}

input:

360

output:

0D 00:06:00
