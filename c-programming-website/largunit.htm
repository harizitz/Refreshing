#include< stdlib.h>

int main()

{

int N,i,d,b,max=0;

scanf("%d",&N);

int a[N];

for(i=0;i< N;i++)

{

scanf("%d ",&a[i]);

}

for(i=0;i< N;i++)

{

d=a[i];

while(d)

{

b=a[i]%10;

if(b>max)

{

max=b;

}

d=d/10;

}

}

for(i=0;i< N;i++)

{

d=a[i]%10;

}

}
