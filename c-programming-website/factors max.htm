Factors count max

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N;

scanf("%d",&N);

int a[N],i,j,c=0,max=0;

for(i=0;i< N;i++)

{

c=0;

scanf("%d",&a[i]);

for(j=1;j<=1000;j++)

{

if(a[i]%j==0)

{

c++;

if(c>max)

{

max=c;

}

}}

}

printf("%d",max);

}

3

10 20 30

output:

8
