First and Second Array reverse Sum

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N;

scanf("%d",&N);

int a[N],i,j,b[N];

for(i=0;i< N;i++)

{

scanf("%d ",&a[i]);

}

for(j=0;j< N;j++)

{

scanf("%d ",&b[j]);

}

for(i=0,j=N-1;i< N,j>=0;i++,j--)

{

printf("%d ",a[i]+b[j]);

}

}

input:

3

1 2 3

5 7 9

output:

10 9 8
