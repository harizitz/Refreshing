Max sum odd and even tenth digit

#include< stdio.h>

#include< stdlib.h>

int main()

{

int N,i,S1=0,S2=0,d,b;

scanf("%d",&N);

int a[N];

for(i=0;i< N;i++)

{

scanf("%d ",&a[i]);

}

for(i=0;i< N;i++)

{

b=(a[i]%100)/10;

if(b%2==1)

{

S1=S1+b;

}

else if(b%2==0)

{

S2=S2+b;

}

}

if(S1>S2)

{

printf("%d",S1);

}

else

{

printf("%d",S2);

}

}

input:

5

10 20 30 40 60

output:

12
